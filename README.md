### 下载
    git clone git://github.com/coldnight/vim_config.git ~/.vim

### 创建vimrc
    ln -sf ~/.vim/vimrc ~/.vimrc

### 安装vundle
    git clone https://github.com/gmarik/vundle.git ~/.vim/bundle/vundle

### 安装插件
    打开Vim,执行末行命令
    :BundleInstall

### 安装扩展
    cd ~/.vim/bundle/jedi-vim/jedi
    sudo python setup.py install
